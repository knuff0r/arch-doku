\subsection{\LaTeX}
Paket: \textbf{texlive-core} \\
\emph{.sty}-files kommen nach \emph{texmf/tex/latex/local}.
Mit 
\begin{lstlisting}[style=Bash]
$ grep -l '\\documentclass' *tex | xargs latexmk -pdf -pvc -silent
\end{lstlisting}
\begin{lstlisting}[style=Bash]
$ texhash
\end{lstlisting}
wird die Latex database aktualisiert.
\dirtree{%
    .1 /.
    .2 texmf.
    .3 tex.
    .4 latex.
    .5 local.
}
